from picamera import PiCamera
from time import sleep # importera allt som behövs

camera = PiCamera()   # skapa kamera
camera.rotation = 180   # sätt rotation 
camera.image_effect ="watercolor" #sätt image effects
camera.brightness = 60 # sätt brightness
camera.color_effects=(128,128)# sätt color effects
camera.awb_mode = "sunlight" # sätter awb mode till sunlight
camera.annotate_text = "Hello world!" # annoterar texten Hello World
camera.annotate_background = picamera.Color('black') # ger annoteringen svart bakgrund
camera.annotate_foreground = picamera.Color('yellow') # ger annoteringen gul förgrund
camera.start_preview(fullscreen = false, window = (x,y,w,h))# starta förhandsgranskning i fönster
for i in range(2): #två varv i loopen
    sleep(3) # sova tre sekunder
    camera.capture('/home/pi/Downloads/testcam5%s.jpg' % i) # lägg till bilden i Downloads mappen
camera.stop_preview()# avsluta förhandsgranskning

Raspistill:

raspistill # tar en bild med raspberry camera

raspistill -o Desktop/image.jpg # -o anger output vilket är Desktop/image.jpg här

raspistill -o Desktop/image-small.jpg -w 640 -h 480 # -w och -h anger dimensioner i pixlar

raspivid -o Desktop/video.h264 # spelar in video

raspistill -vf # -vf flippar bilden vertikalt

raspistill -hf # -hf flippar bilden horisontellt

raspistill -t 2000 # -t anger fördröjning i ms

raspistill -p # -p visar förhandsgranskning (preview)
                #<'x,y,w,h'> för position och dimensioner

raspistill -f # -f visar förhandsgranskning på hela skärmen

raspistill -ex  # -ex anger exposure mode, kan vara:
                #off,auto,night,nightpreview,backlight,spotlight,sports,
                #snow,beach,verylong,fixedfps,antishake,fireworks

raspistill -awb # -awb anger awb mode, kan vara:
                #off,auto,sun,cloud,shade,tungsten,fluorescent,incandescent,
                #flash,horizon

raspistill -ifx # -ifx anger image effect, kan vara:
                #none,negative,solarise,sketch,denoise,emboss,oilpaint,hatch,
                #gpen,pastel,watercolour,film,blur,saturation,colourswap,washedout,
                #posterise,colourpoint,colourbalance,cartoon

raspistill -rot # -rot anger rotation (0-359)

raspistill -br # -br anger ljusstyrka (0-100, där 50 är standard)
